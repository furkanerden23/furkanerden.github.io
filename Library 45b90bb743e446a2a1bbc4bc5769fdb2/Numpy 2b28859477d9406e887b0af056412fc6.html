Basics

1 dimension array: array(x)

import numpy as np                                  # Import

array_1d = np.array([1.1, 9.2, 8.1, 4.7])           # 1 dimensional array, ndarray: n dimensional array

array_1d[2]                                         # Access '2'. index element

array_1d.shape                                      # Element number of array
array_1d.ndim                                       # Dimensions number of array attribute

2 dimension array: array(x, y)

array_2d = np.array([[1, 2, 3, 9], [5, 6, 7, 8]])   # 2d array, 2 rows and 4 columns

array_2d[1, 2]                                      # 7
array_2d[0, :]                                      # Access an entire '0'. index row

array_2d.shape[0]                                   # Rows
array_2d.shape[1]                                   # Columns

array_2d.ndim                                       # Dimensions number of array attribute

3 dimension array: array(x, y, z)

array_3d = np.array([[[0, 1, 2, 3],                 # 3 dimensional array
                      [4, 5, 6, 7]],
                 
                     [[7, 86, 6, 98],
                      [5, 1, 0, 4]],
                     
                     [[5, 36, 32, 48],
                      [97, 0, 27, 18]]])

array_3d[2, 1, 3]                                   # '2'. index group, '1'. index row, '3'. index column: 18
array_3d[2, 1, :]                                   # '2'. index group, '1'. index row, all columns([97, 0, 27, 18])
array_3d[:, :, 0]                                   # '0'. index columns: ([[0, 4], [7, 5], [5, 97]])

array_3d.shape                                      # (3, 2, 4)
array_3d.ndim                                       # Dimensions number of array attribute

Create arrays

a = np.arange(10,30)                                # All numbers between '10' - '30', '30' not incuded

Array slice

a[:6]                                               # From first to '6'.
a[3:6]                                              # From '3'. to '6'., '6'. not included
a[12:]                                              # From '12'. to last
a[-3:]                                              # From '-3 (3rd from last)' to last
a[::2]                                              # From first to last with '2' steps

np.flip(a)                                          # reverse the order of an array
a[::-1]                                             # reverse the order of an array

Zero values in array

b = np.array([6,0,9,0,0,5,0])                       # An array has zero values
nonzero_indices = np.nonzero(b)                     # Delete zero values and this returns a tuple

Array Calculations

1 dimension array calculations

v1 = np.array([4, 5, 2, 7])
v2 = np.array([2, 1, 3, 3])

v1 + v2                                         # Result: array([ 6, 6, 5, 10])
v1 * v2                                         # Result: array([ 8, 5, 6, 21])
v1 + 10                                         # Result: array([14, 15, 12, 17]) 
v1 / v2                                         # Result: array([2, 5, 0.66, 2.33])

2 dimension array multiplication

a1 = np.array([[1, 3],                          # 4 x 2 matrix
               [0, 1],
               [6, 2],
               [9, 7]])
 
b1 = np.array([[4, 1, 3],                       # 3 x 2 matrix
               [5, 8, 5]])

c = np.matmul(a1, b1)                           # Multiply 2 different matrix
a1 @ b1                                         # This is also gives the same result

Images with Array

Set-up

from scipy import misc                          # This has a image of racoon to work on it
from PIL import Image                           # This is for reading image files
import matplotlib.pyplot as plt                 # Display the image

img = misc.face()                               # This is a ndarray, values are between 0-255(RGB)
plt.imshow(img)                                 # Show the image

Basics

type(img)                                       # 'numpy.ndarray'
img.shape                                       # (768, 1024, 3)
img.ndim                                        # 3

Turn the image grayscale

sRGB_array = img / 255                          # Every value becomes between 0-1(sRGB)
grey_vals = np.array([0.2126, 0.7152, 0.0722])  # Gray scale multiplication, multiply it with sRGB values
img_gray = sRGB_array @ grey_vals               # img_gray turns to grayscale values
plt.imshow(img_gray, cmap='gray')               # Show this image

Flip, rotate and solar image

np.flip(img)                                    # Flips image in x and y axis

np.rot90(img)                                   # Rotate image 90 degrees

solar_img = 255 - img                           # Solar image (negative)

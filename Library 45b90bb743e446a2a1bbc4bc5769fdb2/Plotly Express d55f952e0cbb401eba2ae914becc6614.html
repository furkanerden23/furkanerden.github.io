import plotly.express as px
import pandas

data = pandas.read_csv('Resources\weather_data.csv')

Graphs

line = px.line(data, 
               x='date', 
               y='temp',
               title='Temperature Day by Day',
               range_x=['2022-9-1','2022-9-23'],
               range_y=[10, 26])

line.update_layout(xaxis_title='Dates',  
                   yaxis_title='Temperatures',
                   xaxis={'categoryorder':'min ascending'})

line.show()
figure = px.pie(labels=data.date,                       # Rows
                values=data.temp,                       # Data values
                title='Weather Data',       
                names=data.day,                         # Name of slices of pie chart
                hole=0.6)       
        
        
figure.update_traces(textposition='outside',            # Text of slices, display name in outside of chart
                     textfont_size=15,                  # Text size of chart
                     textinfo='percent+label')          # Text configuration

figure.show()
figure = px.bar(x = data.temp,
                y = data.date,
                orientation='h',
                title='Weather Data')

figure.update_layout(xaxis_title='Temperatures',  
                     yaxis_title='Dates',
                     yaxis={'categoryorder':'min ascending'})
figure.show()
g_bar = px.bar(data,
               x='date',
               y='temp',
               title='Free vs Paid Apps by Category',
               color='height')

g_bar.update_layout(xaxis_title='Date',  
                     yaxis_title='Temperatures',
                     xaxis={'categoryorder':'min ascending'},
                     coloraxis_showscale=True)              # Show colorscale right side
g_bar.show()
g_bar = px.bar(data,
               x='day',
               y='temp',
               title='Temperature vs Condition by Days',
               color='condition',
               barmode='group')

g_bar.update_layout(xaxis_title='Day',
                    yaxis_title='Temperature',
                    xaxis={'categoryorder':'total descending'},
                    yaxis=dict(type='log'))

g_bar.show()
figure = px.scatter(data,                                   # Data
                    x='date',                               # x values
                    y='temp',                               # y values
                    title='Weather Data',                   # Title of chart
                    size='temp',                            # What is gonna decide to size of chart
                    hover_name=data.day,                    # What we gonna see when we drag mouse to a dot on chart
                    color='height',                         # Change color of dots based on that
                    color_continuous_scale='Agsunset')      # Color continuous over dots

figure.update_layout(xaxis_title='Days',  
                     yaxis_title='Temperatures',
                     xaxis={'categoryorder':'min ascending'},
                     coloraxis_showscale=True)              # Show colorscale right side

figure.show()
box = px.box(data,
             x='condition',
             y="temp",
             title='Temp on Condition')
 
box.update_layout(xaxis_title='Condition',
                  yaxis_title='Temperature',
                  xaxis={'categoryorder':'max descending'},
                  yaxis=dict(type='log'))
 
box.show()
world_map = px.choropleth(data,                                                     # Create a world map to show data
                          locations='ISO',                                          # 'ISO' country codes for data place on map
                          color='temp',                                             # Country colors gonna change based on this
                          hover_name='ISO',                                         # Show this when mouse drag on country
                          color_continuous_scale=px.colors.sequential.matter)       # Color scale

world_map.show()
burst = px.sunburst(data,                                                                       # Create sunburst chart
                    path=['condition', 'height', 'ISO'],    # Columns of values
                    values='temp',                                                             # Chart is gonna created based on that
                    title='Weather',
                   )

burst.show()
hist = px.histogram(data, 
                    x='temp', 
                    color='condition',
                    nbins=30,
                    opacity=0.6,
                    barmode='overlay',
                    histnorm='percent',
                    marginal='box')

hist.update_layout(xaxis_title='Temps',  
                   yaxis_title='Percentage',
                   xaxis={'categoryorder':'min ascending'},
                   coloraxis_showscale=False)                     # Makes the colour axis disappear

hist.show()

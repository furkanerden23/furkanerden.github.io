Features

list_str.append('David')                        # Append 'David' to end of list
list_str.clear()                                # Clear that list
list_str.copy()                                 # Copy and return that copied element
list_str.count('John')                          # How many 'John's are there in list
list_str.extend(list_int)                       # Append 'list_int' to 'list_str'
list_str.index('Peter')                         # Index of 'Peter' in list
list_str.insert(3, 'George')                    # Append 'George' to '3'. index
list_str.pop(2)                                 # Delete '2'. index; if 'pop()' deletes last index
list_str.remove('Vicky')                        # Remove 'Vicky' from list
list_str.reverse()                              # Reverse list
list_str.sort()                                 # Sort list

len(list_int)                                   # Length of list
max(list_int)                                   # Highest element of list
min(list_int)                                   # Lowest element of list
sum(list_int)                                   # Summary of list

List Slice

piano_keys = ["a", "b", "c", "d", "e", "f", "g"]

print(piano_keys[2:5])                          # From '2.' to '5'.             "c", "d", "e"
print(piano_keys[2:])                           # From '2.' to last             "c", "d", "e", "f", "g"
print(piano_keys[:5])                           # From first to '5'.            "a", "b", "c", "d", "e"
print(piano_keys[2:5:2])                        # From '2.' to '5'. '2' steps   "c", "e"
print(piano_keys[::2])                          # '2' steps                     "a", "c", "e", "g"
print(piano_keys[::-1])                         # Reverse the list

List Comprehension

numbers = [1, 2, 3]
new_list = [n + 1 for n in numbers]             # Gets every number from 'numbers' +1 to new list

name = "John"
letters_list = [letter for letter in name]      # Gets the letters as list from the name

range_list = [num * 2 for num in range(1, 5)]   # You can create a list from a range

names = ["Alex", "Beth", "Caroline", "Dave", "Eleanor", "Freddie"]
short_names = [name for name in names if len(name) < 5]     # Only get the names shorter than 5 letters

data = {}
data_list = [value for (key, value) in data.items()]        # Turn a dic into a list without keys

Dictionary Comprehension

names = ["Alex", "Beth", "Caroline", "Dave", "Eleanor", "Freddie"]
students_scores = {student: random.randint(1, 100) for student in names}    # Generate random scores for dictionary

passed_students = {student: score for (student, score) in students_scores.items() if score >= 60}  # Loop a dictionary

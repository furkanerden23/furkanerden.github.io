Guess the Number

Introduction

This game is fun. Computer holds a random number between 1-100. You are trying to guess that number. You have 5 attempts for hard mode, 10 for easy. Game tells you if your guess is low or high. You should just find the number.

The aim of this project

This project requires comparisons. Compare the number chosen by computer and player input. Direct the player to the right way: upwards or backwards. Meanwhile, check the remaining attempt number.

Achievements
  • Use functions
  • While/for loops
  • If statements
  • Do comparisons
  • Input function usage
  • Store data in variables
  • Print information with F string

// Function to generate a random number between min and max (inclusive)
function getRandomNumber(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

const EASY_LEVEL_TURNS = 10;
const HARD_LEVEL_TURNS = 5;

// Function to check user's guess against actual answer.
function checkAnswer(guess, answer, turns) {
    // Checks answer against guess. Returns the number of turns remaining.
    if (guess > answer) {
        alert("Too high.");
        return turns - 1;
    } else if (guess < answer) {
        alert("Too low.");
        return turns - 1;
    } else {
        alert(`You got it! The answer was ${answer}.`);
        return turns;
    }
}

// Function to set difficulty.
function setDifficulty() {
    const level = prompt("Choose a difficulty. Type 'easy' or 'hard': ").toLowerCase();
    if (level === "easy") {
        return EASY_LEVEL_TURNS;
    } else {
        return HARD_LEVEL_TURNS;
    }
}

function game() {
    // Choosing a random number between 1 and 100.
    alert("Welcome to the Number Guessing Game!");
    alert("I'm thinking of a number between 1 and 100.");
    const answer = getRandomNumber(1, 100);

    let turns = setDifficulty();
    let guess = 0;

    // Repeat the guessing functionality if they get it wrong.
    while (guess !== answer) {
        alert(`You have ${turns} attempts remaining to guess the number.`);
        // Let the user guess a number.
        guess = parseInt(prompt("Make a guess: "));

        // Track the number of turns and reduce by 1 if they get it wrong.
        turns = checkAnswer(guess, answer, turns);
        if (turns === 0) {
            alert("You've run out of guesses, you lose.");
            alert(`The answer was ${answer}.`);
            return;
        } else if (guess !== answer) {
            alert("Guess again.");
        }
    }
}

game();

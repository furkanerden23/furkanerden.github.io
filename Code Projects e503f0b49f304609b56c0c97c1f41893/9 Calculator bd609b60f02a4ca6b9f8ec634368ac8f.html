Calculator

Introduction

Let’s create our own calculator. It’s gonna be able to do basic operations: addition, subtraction, multiplication and division.

The aim of this project

Most important thing is being able to call different functions with the same variable. If the user wants to do an multiplication, it’s gonna call ‘multiply’ function, if the user wants to do a division, it’s gonna call ‘divide’ function without doing a change inside of the code.

Achievements
  • Call functions with a variable
  • Use functions
  • While/for loops
  • If statements
  • Input function usage
  • Store data in variables
  • Print information with F string

Javascript code
function add(n1, n2) {
  return n1 + n2;
}

function subtract(n1, n2) {
  return n1 - n2;
}

function multiply(n1, n2) {
  return n1 * n2;
}

function divide(n1, n2) {
  return n1 / n2;
}

const operations = {
  "+": add,
  "-": subtract,
  "*": multiply,
  "/": divide
};

function calculator() {
  let num1 = parseFloat(prompt("What's the first number?: "));
  let operationSymbols = "";
  for (let symbol in operations) {
    operationSymbols += symbol + "\n";
  }
  alert(operationSymbols);
  let shouldContinue = true;

  while (shouldContinue) {
    let operationSymbol = prompt("Pick an operation: ");
    let num2 = parseFloat(prompt("What's the next number?: "));
    let calculationFunction = operations[operationSymbol];
    let answer = calculationFunction(num1, num2);
    alert(num1 + " " + operationSymbol + " " + num2 + " = " + answer);

    if (prompt("Type 'y' to continue calculating with " + answer + ", or type 'n' to start a new calculation: ") === 'y') {
      num1 = answer;
    } else {
      shouldContinue = false;
      calculator();
    }
  }
}

calculator();

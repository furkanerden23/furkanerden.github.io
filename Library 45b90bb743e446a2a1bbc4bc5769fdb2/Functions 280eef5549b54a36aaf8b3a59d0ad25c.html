Default Values

    return a + b + c

Global Scope

one = 1                                         # Functions can reach but can't change

def global_scope():
    global one                                  # This function can now change
    two = 2                                     # Global and other functions can't reach
    pass

Hints and Arrows

*args

def arg_func(*args):                            # Unlimited inputs with '*args'
    args[0]                                     # You can call any arg
    for n in args:
        pass
    
arg_func(1, 2, 3, 4, 5)                         # Just like a list

**kwargs

def kwargs_func(**kwargs):                      # Input with keyword, like a dictionary
    kwargs['add']
    kwargs.get['subs']                          # This is NOT ERROR if 'subs' keyword is not given as input
    for key, value in kwargs.items():
        pass
    
kwargs_func(add=1, subs=2)

Functions as Inputs

def calculate(calc_func, a, b):
    return calc_func(a, b)

calculate(func, 1, 2)                           # 'func' comes from 'Basics' section

Inner Function and Return Function

def outher():
    def inner(a, b):
        return a + b
    return inner                                # Returns inner function but doesn't run it

self = outher()                                 # 'self' is now becomes 'inner()'
self(1, 2)

Decorator Functions

def decorator_func(func):
    def wrapper_func():
        pass                                    # Decorations here
        func()
    return wrapper_func

@decorator_func                                 # decorators_func(add), wrapper_func()
def add():
    return

add()                                           # Call function like normal function

Decorators with Args and Kwargs

def decorators_func(func):
    def wrapper_func(*args, **kwargs):          # Args comes from decorated func input
        args[0]
        pass
        func(args)                              # Give arguments back to work with it
    return wrapper_func

@decorators_func                                # decorators_func(add), wrapper_func(a) executed
def add(a):
    print(a)
    return

add(1)
